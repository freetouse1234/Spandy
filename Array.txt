#include <stdio.h>
int main(){
	
	int arr[15];
	printf("Enter the elements in the array\n");
	for (int i=0;i<15;i++)
	{
		int ele;
		scanf("%d",&ele);
		arr[i]=ele;
	}
	
	printf("The elements in the array are: \n");
	for (int i=0;i<15;i++)
	{
		printf("%d\t",arr[i]);
	}

	return 0;

}
